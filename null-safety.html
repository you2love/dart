空安全

可空类型

可空类型

null_safety.dart
void main() {
  String name = 'Dart';  // 非空
  String? nullableName;    // 可空
  nullableName = null;
  print(name?.length);
  print(name ?? 'default');
}

空安全操作符

operators.dart
void main() {
  String? name;
  print(name?.length);      // null
  print(name ?? 'default'); // default
  name ??= 'Dart';
  print(name);              // Dart
}
