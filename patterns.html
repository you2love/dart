模式匹配

switch

switch 表达式

switch_expr.dart
void main() {
  var day = 3;
  var dayName = switch (day) {
    1 => 'Monday',
    2 => 'Tuesday',
    3 => 'Wednesday',
    _ => 'Other',
  };
  print(dayName); // Wednesday
}

记录类型

records.dart
void main() {
  var user = (name: 'Alice', age: 30);
  print(user.name); // Alice
  var (name, age) = user;
  print('$name is $age'); // Alice is 30
}
