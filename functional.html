higher_order.dart
void main() {
  var numbers = [1, 2, 3, 4, 5];
  var doubled = numbers.map((n) => n * 2).toList();
  print(doubled); // [2, 4, 6, 8, 10]
  var evens = numbers.where((n) => n % 2 == 0).toList();
  print(evens); // [2, 4]
  var sum = numbers.fold(0, (a, b) => a + b);
  print(sum); // 15
}
