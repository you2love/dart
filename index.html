开始学习

为什么学习 Dart？

Dart 是由 Google 开发的现代编程语言，主要用于构建 Web、移动端和桌面应用。它是 Flutter 框架的官方语言，具有以下特点：

🚀

高性能

JIT 和 AOT 双编译模式，开发时快速热重载，生产时编译为高效的本地代码

🔒

空安全

编译时静态检查空值引用，消除空指针异常，提高代码可靠性

🎯

类型安全

强大的类型系统，支持泛型、类型推断，同时保持代码简洁

🌐

跨平台

一套代码同时运行在 iOS、Android、Web、Windows、Linux、macOS

  1. 访问 dartpad.dev（在线编辑器，无需安装）

扩展方法

无需继承即可为现有类型添加方法：

extensions.dart 
// 扩展 String
extension StringExtensions on String {
  bool get isEmail {
    return RegExp(r'^[\w-\.]+@[\w-]+\.[\w-]{2,4}$').hasMatch(this);
  }

  String capitalize() {
    if (isEmpty) return this;
    return this[0].toUpperCase() + substring(1);
  }

  String get reversed => split('').reversed.join();
}

// 扩展 List
extension ListExtensions on List {
  List get shuffled => toList()..shuffle();

  T? get firstOrNull => isEmpty ? null : first;

  List> chunk(int size) {
    final chunks = >[];
    for (var i = 0; i < length; i += size) {
      chunks.add(sublist(i, i + size > length ? length : i + size));
    }
    return chunks;
  }
}

void main() {
  print('test@example.com'.isEmail);     // true
  print('hello'.capitalize());           // Hello
  print('Dart'.reversed);                // traD

  print([1, 2, 3, 4].chunk(2));         // [[1, 2], [3, 4]]
  print([].firstOrNull);                 // null
}

构造函数的技巧

constructors.dart 
class Point {
  final double x;
  final double y;

  // 命名参数 + 默认值
  Point({this.x = 0, this.y = 0});

  // 命名构造函数
  Point.origin() : x = 0, y = 0;
  Point.unit() : x = 1, y = 1;

  // 重定向构造函数
  Point.zero() : this(x: 0, y: 0);

  // Factory 构造函数（可返回子类或缓存实例）
  factory Point.fromMap(Map map) {
    return Point(x: map['x'] ?? 0, y: map['y'] ?? 0);
  }

  // 复制并修改
  Point copyWith({double? x, double? y}) {
    return Point(x: x ?? this.x, y: y ?? this.y);
  }

  @override
  String toString() => 'Point($x, $y)';
}

class Rectangle {
  final Point topLeft;
  final Point bottomRight;

  Rectangle({required this.topLeft, required this.bottomRight});

  // 命名构造函数 + 初始化列表
  Rectangle.fromSize(double width, double height)
      : topLeft = Point(),
        bottomRight = Point(x: width, y: height);

  double get width => bottomRight.x - topLeft.x;
  double get height => bottomRight.y - topLeft.y;
}

void main() {
  var p1 = Point(x: 3, y: 4);
  var p2 = Point.origin();
  var p3 = Point.fromMap({'x': 5, 'y': 6});

  var p4 = p1.copyWith(x: 10);
  print(p4);                          // Point(10.0, 4.0)
}

空安全

可空类型、空操作符

可空类型

Dart 2.12 引入空安全，类型默认不可为空：

null_safety.dart 
void main() {
  // 非空类型（不能为 null）
  String name = 'Dart';
  // name = null;  // 编译错误！

  // 可空类型（添加 ?）
  String? nullableName;
  nullableName = 'Dart';
  nullableName = null;  // 合法

  // 访问可空类型的方法需要空安全操作符
  String? text;
  // text.length;      // 编译错误
  print(text?.length); // null

  // late 延迟初始化
  late String lateInit;
  // print(lateInit);  // 运行时错误（未初始化）
  lateInit = 'initialized';
  print(lateInit);      // initialized

  // 可空集合
  List? nullableList;
  nullableList = [1, 2, 3];
  nullableList = null;

  // 非空集合，但元素可空

}

Future

Future 代表一个异步操作的最终结果：

future.dart 
// 模拟异步操作
Future fetchUser() async {
  await Future.delayed(Duration(seconds: 1));
  return 'User Data';
}

// 错误处理
Future parseNumber(String input) async {
  await Future.delayed(Duration(milliseconds: 500));
  return int.parse(input);  // 可能抛出异常
}

void main() async {
  // async/await
  print('Fetching user...');
  var user = await fetchUser();
  print('Got: $user');

  // Future.then
  fetchUser().then((data) => print('Then: $data'));

  // 错误处理
  try {
    var num = await parseNumber('42');
    print('Parsed: $num');
  } catch (e) {
    print('Error: $e');
  }

  // 多个异步操作
  var results = await Future.wait([
    Future.delayed(Duration(milliseconds: 100), () => 'a'),
    Future.delayed(Duration(milliseconds: 200), () => 'b'),
    Future.delayed(Duration(milliseconds: 150), () => 'c'),
  ]);
  print(results);  // [a, b, c]

  // Future.value - 创建已完成的 Future
  var completed = Future.value('done');
  print(await completed);

  // Future.error - 创建错误的 Future
  var error = Future.error(Exception('Oops'));
}

Stream

Stream 是异步数据序列：

stream.dart 
// 创建 Stream
Stream countStream(int max) async* {
  for (int i = 1; i <= max; i++) {
    await Future.delayed(Duration(seconds: 1));
    yield i;
  }
}

// 监听 Stream
void main() {
  // listen
  countStream(3).listen(
    (value) => print('Received: $value'),
    onDone: () => print('Done'),
    onError: (error) => print('Error: $error'),
  );

  // 转换操作
  var numbers = Stream.fromIterable([1, 2, 3, 4, 5]);

  // map - 转换
  var doubled = numbers.map((n) => n * 2);

  // where - 过滤
  var evens = numbers.where((n) => n % 2 == 0);

  // take - 限制数量
  var firstThree = numbers.take(3);

  // fold - 聚合
  numbers.fold(0, (sum, n) => sum + n).then(print);

  // 链式操作
  Stream.fromIterable([1, 2, 3, 4, 5])
    .where((n) => n > 2)
    .map((n) => n * 10)
    .listen(print);

  // asyncExpand - 扁平化
  Stream.periodic(Duration(seconds: 1), (i) => i)
    .take(3)
    .asyncExpand((i) => Stream.fromIterable([i, i]))
    .listen(print);
}

Isolate

Isolate 是 Dart 的并发原语，用于 CPU 密集型任务：

isolate.dart 
import 'dart:isolate';

// 在独立 Isolate 中运行
Future heavyComputation() async {
  // 创建 ReceivePort 用于通信
  final receivePort = ReceivePort();

  // 在新 Isolate 中运行
  await Isolate.spawn(
    _heavyTask,
    [receivePort.sendPort, 1000000000],
  );

  // 等待结果
  final result = await receivePort.first;
  print('Result: $result');
}

void _heavyTask(List args) {
  final sendPort = args[0] as SendPort;
  final n = args[1] as int;

  // CPU 密集型计算
  var sum = 0;
  for (var i = 0; i < n; i++) {
    sum += i;
  }

  // 发送结果回主 Isolate
  sendPort.send(sum);
}

// Compute 简化方式
import 'dart:isolate';

Future computeSum(int n) async {
  return await Isolate.run(() {
    var sum = 0;
    for (var i = 0; i < n; i++) {
      sum += i;
    }
    return sum;
  });
}

void main() async {
  await heavyComputation();
  var sum = await computeSum(1000000);
  print('Sum: $sum');
}

集合操作

List、Map、Set 提供丰富的函数式操作：

collections.dart 
void main() {
  var numbers = [1, 2, 3, 4, 5];

  // map - 转换
  var doubled = numbers.map((n) => n * 2);
  print(doubled.toList());  // [2, 4, 6, 8, 10]

  // where - 过滤
  var evens = numbers.where((n) => n % 2 == 0);
  print(evens.toList());    // [2, 4]

  // expand - 扁平化
  var nested = [[1, 2], [3, 4]];
  var flat = nested.expand((list) => list);
  print(flat.toList());     // [1, 2, 3, 4]

  // fold - 聚合
  var sum = numbers.fold(0, (a, b) => a + b);
  print(sum);              // 15

  // reduce - 聚合（不需要初始值）
  var product = numbers.reduce((a, b) => a * b);
  print(product);           // 120

  // any / every - 断言
  print(numbers.any((n) => n > 3));   // true
  print(numbers.every((n) => n > 0)); // true

  // firstWhere / singleWhere
  print(numbers.firstWhere((n) => n > 3));  // 4

  // take / skip
  print(numbers.take(3).toList());    // [1, 2, 3]
  print(numbers.skip(2).toList());    // [3, 4, 5]

  // List.generate
  var squares = List.generate(5, (i) => i * i);
  print(squares);                     // [0, 1, 4, 9, 16]

  // Map 操作
  var map = {'a': 1, 'b': 2, 'c': 3};
  var doubledMap = map.map((k, v) => MapEntry(k, v * 2));
  print(doubledMap);                  // {a: 2, b: 4, c: 6}
}

闭包

闭包可以捕获外部变量：

closures.dart 
// 闭包示例
Function makeAdder(int addBy) {
  return (int i) => i + addBy;
}

// 工厂闭包
class Counter {
  int _count = 0;

  int Function() increment() {
    return () => ++_count;
  }
}

// 闭包中的可变状态
Function makeCounter() {
  var count = 0;
  return () {
    count++;
    return count;
  };
}

void main() {
  var add2 = makeAdder(2);
  var add10 = makeAdder(10);

  print(add2(3));    // 5
  print(add10(3));   // 13

  // 闭包捕获变量
  var functions = [];
  for (var i = 0; i < 3; i++) {
    functions.add(() => i);
  }

  for (var f in functions) {
    print(f());      // 0, 1, 2（正确）
  }

  // 工厂闭包
  var counter = Counter();
  var inc = counter.increment();
  print(inc());      // 1
  print(inc());      // 2
}

switch 表达式

Dart 3.0+ 支持强大的 switch 表达式：

switch_expr.dart 
void main() {
  // 基本 switch 表达式
  var day = 3;
  var dayName = switch (day) {
    1 => 'Monday',
    2 => 'Tuesday',
    3 => 'Wednesday',
    _ => 'Other day',
  };
  print(dayName);  // Wednesday

  // 多值匹配
  var num = 15;
  var result = switch (num) {
    1 || 2 || 3 => 'Small',
    4 || 5 || 6 => 'Medium',
    _ => 'Large',
  };

  // 带条件的匹配（when）
  var score = 85;
  var grade = switch (score) {
    >= 90 => 'A',
    >= 80 => 'B',
    >= 70 => 'C',
    >= 60 => 'D',
    _ => 'F',
  };
  print(grade);  // B

  // 记录类型解构
  var point = (x: 3, y: 4);
  var location = switch (point) {
    (x: 0, y: 0) => 'Origin',
    (x: var x, y: 0) => 'On X axis at $x',
    (x: 0, y: var y) => 'On Y axis at $y',
    (x: var x, y: var y) => 'Point($x, $y)',
  };
  print(location);  // Point(3, 4)
}

记录类型

记录是轻量级的数据结构：

records.dart 
void main() {
  // 记录定义
  var user = (name: 'Alice', age: 30, email: 'alice@example.com');

  // 访问字段
  print(user.name);   // Alice
  print(user.$1);     // Alice（位置访问）

  // 解构
  var (name, age, email) = user;
  print('$name is $age years old');

  // 命名解构
  var (name: n, age: a) = user;

  // 记录中的模式匹配
  var point = (x: 10, y: 20);
  switch (point) {
    case (x: 0, y: 0):
      print('Origin');
    case (x: var x, y: var y) when x == y:
      print('On diagonal at $x');
    case (x: var x, y: var y):
      print('Point($x, $y)');
  }

  // 泛型记录
  var box = (value: 42, label: 'answer');
  print(box.value);  // 42
}

对象模式匹配

使用对象进行模式匹配：

object_patterns.dart 
class Point {
  final int x;
  final int y;
  Point(this.x, this.y);
}

class Circle {
  final Point center;
  final double radius;
  Circle(this.center, this.radius);
}

void main() {
  var circle = Circle(Point(0, 0), 5);

  // 对象模式匹配
  var description = switch (circle) {
    Circle(center: Point(x: 0, y: 0), radius: var r) =>
      'Circle at origin with radius $r',
    Circle(center: Point(x: var x, y: var y), radius: var r) =>
      'Circle at ($x, $y) with radius $r',
  };
  print(description);

  // if-case 语句
  var point = Point(3, 4);
  if (point case Point(x: var x, y: var y) when x == y) {
    print('Point is on diagonal');
  } else {
    print('Point is not on diagonal');
  }

  // for 循环中的模式
  var points = [Point(1, 1), Point(2, 2), Point(3, 4)];
  for (Point(x: var x, y: var y) in points) {
    print('Point($x, $y)');
  }
}

生成器

同步和异步生成器：

generators.dart 
// 同步生成器 - 使用 yield
Iterable naturalsTo(int n) sync* {
  var k = 0;
  while (k < n) {
    yield k++;
  }
}

// 递归生成器
Iterable fibonacci(int n) sync* {
  if (n > 0) yield 1;
  if (n > 1) yield 1;
  var a = 1, b = 1;
  for (var i = 3; i <= n; i++) {
    var c = a + b;
    yield c;
    a = b;
    b = c;
  }
}

// 异步生成器 - 使用 async* 和 yield
Stream countDown(int from) async* {
  for (var i = from; i > 0; i--) {
    await Future.delayed(Duration(seconds: 1));
    yield i;
  }
}

// yield* 委托
Iterable naturalsFrom(int start, int end) sync* {
  yield* Iterable.generate(end - start, (i) => start + i);
}

void main() {
  // 使用同步生成器
  for (var n in naturalsTo(5)) {
    print(n);  // 0, 1, 2, 3, 4
  }

  print(fibonacci(7).toList());  // [1, 1, 2, 3, 5, 8, 13]

  // 使用异步生成器
  countDown(3).listen((n) => print('Countdown: $n'));
}

延迟加载

延迟成员和库的懒加载：

lazy.dart 
// late 延迟初始化
class WeatherService {
  late final String _apiKey;

  WeatherService() {
    // 构造函数中不初始化，延迟到首次访问
  }

  void init(String key) {
    _apiKey = key;
  }

  String get apiKey => _apiKey;
}

// 使用 late 避免空检查
class User {
  final String name;
  late final String email;  // 稍后设置

  User(this.name);

  void setEmail(String e) {
    email = e;
  }

  void printEmail() {
    // 无需空检查，因为 late 保证会被初始化
    print(email.toLowerCase());
  }
}

// get延迟计算
class Circle {
  final double radius;

  Circle(this.radius);

  // 每次访问都重新计算
  double get area => 3.14159 * radius * radius;

  // 延迟计算（使用 late）
  late final double _cachedArea = 3.14159 * radius * radius;
  double get cachedArea => _cachedArea;
}

// 库延迟加载
// import 'package:lib/lib.dart' deferred as lib;

  var circle = Circle(5);
  print(circle.area);         // 每次计算
  print(circle.cachedArea);    // 只计算一次
}

Mixins 进阶

Mixin 是代码复用的强大方式：

mixins.dart 
// 基础 Mixin
mixin Logger {
  void log(String msg) => print('[LOG] $msg');
}

// 带状态的 Mixin
mixin Counter {
  int _count = 0;
  int get count => _count;
  void increment() => _count++;
}

// 带约束的 Mixin（使用 on 限定父类）
abstract class Animal {
  void speak();
}

mixin CanFly on Animal {
  void fly() => print('Flying!');
}

mixin CanSwim on Animal {
  void swim() => print('Swimming!');
}

// 使用 with 应用多个 Mixin
class Bird extends Animal with CanFly {
  @override
  void speak() => print('Tweet!');
}

class Duck extends Animal with CanFly, CanSwim {
  @override
  void speak() => print('Quack!');
}

// Mixin 中的 super 调用
mixin onComparable {
  int compareTo(T other);

  bool operator <(T other) => compareTo(other) < 0;
  bool operator >(T other) => compareTo(other) > 0;
}

class Version implements Comparable {
  final int major;
  final int minor;

  Version(this.major, this.minor);

  @override
  int compareTo(Version other) {
    if (major != other.major) return major.compareTo(other.major);
    return minor.compareTo(other.minor);
  }
}

void main() {
  var bird = Bird();
  bird.speak();  // Tweet!
  bird.fly();    // Flying!

  var duck = Duck();
  duck.fly();    // Flying!
  duck.swim();  // Swimming!
}
